[cm5]
dtoverlay=dwc2,dr_mode=host

# Peripheral mode for the USB gadget (UVC webcam or CDC-NCM, one at a time —
# see rpi-cam-gadget-mode.sh). Only boards with a dedicated OTG port get this;
# Pi 3/3+ route USB through an on-board hub (host-only) and are omitted.
#
# dr_mode=peripheral (NOT otg): the device is always a pure gadget, so there is
# no host role to negotiate. With otg, dwc2 leans on the (floating) micro-USB ID
# pin to pick a role and intermittently lands in host mode — we observed a
# "Mode Mismatch Interrupt" storm and failed/full-speed enumeration. peripheral
# is deterministic and removes that ambiguity. (The earlier note blaming
# peripheral for a full-speed/descriptor-timeout was a misdiagnosis: that was
# the UVC+NCM *composite* over-running the dwc2 FIFO, not the dr_mode.)
[pi0]
# Pi Zero / Zero W — micro-USB is the OTG port.
dtoverlay=dwc2,dr_mode=peripheral

[pi02]
# Pi Zero 2 W — micro-USB is the OTG port.
dtoverlay=dwc2,dr_mode=peripheral

[pi4]
# Pi 4 USB-C.
dtoverlay=dwc2,dr_mode=peripheral
