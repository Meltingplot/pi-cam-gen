[cm4]

otg_mode=1

[cm5]
dtoverlay=dwc2,dr_mode=host

# USB gadget: composite UVC webcam + CDC-NCM network (see rpi-cam-gadget-setup.sh).
# Only boards with a dedicated OTG port get this; Pi 3/3+ route USB through an
# on-board hub (host-only) and are omitted.
#
# dr_mode=otg (NOT peripheral): empirically the Zero 2 W's dwc2 fails to bring
# up the high-speed PHY in forced-peripheral mode — the device comes up
# full-speed with a dead ep0 and the host's descriptor read times out
# (DevSpd=full-speed in DSTS). otg does the full OTG bring-up and resolves to
# peripheral when attached to a host, and the gadget enumerates at high speed.
# (The "Mode Mismatch Interrupt" lines seen in otg are benign.)
[pi0]
# Pi Zero family — micro-USB is the OTG port. [pi0] matches Zero, Zero W AND
# Zero 2 W, so there is intentionally NO separate [pi02] section: adding one
# would make the firmware apply the dwc2 overlay twice on the Zero 2 W.
dtoverlay=dwc2,dr_mode=otg

[pi4]
# Pi 4 USB-C.
dtoverlay=dwc2,dr_mode=otg
